/* ── EXCEPTION PARSER ── */
function parseException(txt){
  if(!txt) return [];
  var results=[], seen={};
  var patterns=[
    {re:/TypeError/i,           type:'TypeError',         sev:'critical',hint:'Accessing a property on undefined or null. Ensure the object exists before use.'},
    {re:/ReferenceError/i,      type:'ReferenceError',    sev:'critical',hint:'Variable is not defined. Check scope and declaration order.'},
    {re:/SyntaxError/i,         type:'SyntaxError',       sev:'critical',hint:'Code has a syntax problem. Check for missing brackets, commas, or typos.'},
    {re:/NullPointerException/i,type:'NullPointerException',sev:'critical',hint:'Java null dereference. Ensure the object is initialized before use.'},
    {re:/OutOfMemoryError/i,    type:'OutOfMemoryError',  sev:'critical',hint:'Heap exhausted. Profile memory and look for leaks.'},
    {re:/StackOverflow/i,       type:'StackOverflowError',sev:'critical',hint:'Infinite recursion. Check base conditions in recursive functions.'},
    {re:/HTTP\s*5\d\d/i,        type:'HTTP 5xx Error',    sev:'critical',hint:'Server error. Check server logs, DB connections, and service health.'},
    {re:/HTTP\s*4\d\d/i,        type:'HTTP 4xx Error',    sev:'warning', hint:'Client error. Check request headers, auth tokens, and endpoint URLs.'},
    {re:/CORS/i,                type:'CORS Error',        sev:'warning', hint:'Cross-origin blocked. Check allowed-origins in API gateway config.'},
    {re:/timeout/i,             type:'Timeout',           sev:'warning', hint:'Request timed out. Check network latency and DB query performance.'},
    {re:/403/,                  type:'403 Forbidden',     sev:'warning', hint:'Authorization denied. Verify user roles and permissions.'},
    {re:/404/,                  type:'404 Not Found',     sev:'info',    hint:'Resource not found. Check route config and data existence.'}
  ];
  patterns.forEach(function(pat){
    if(pat.re.test(txt) && !seen[pat.type]){
      seen[pat.type]=true;
      results.push({type:pat.type, sev:pat.sev, hint:pat.hint});
    }
  });
  return results;
}

function doCompare(){
  var idA=$('cmp-a').value, idB=$('cmp-b').value;
  if(idA===idB){toast('Select two different reports');return;}
  var A=HIST.find(function(h){return h.id==idA;}), B=HIST.find(function(h){return h.id==idB;});
  if(!A||!B) return;
  var SM={title:{l:'Title',m:20},description:{l:'Desc',m:20},steps:{l:'Steps',m:20},expected_actual:{l:'Exp/Act',m:15},environment:{l:'Env',m:15},severity:{l:'Severity',m:5},attachments:{l:'Attach',m:5}};
  var winner=A.score>=B.score?'A':'B';
  var bars='';
  Object.keys(SM).forEach(function(k){
    var va=(A.parsed&&A.parsed.section_scores?A.parsed.section_scores[k]:0)||0;
    var vb=(B.parsed&&B.parsed.section_scores?B.parsed.section_scores[k]:0)||0;
    var pa=Math.round((va/SM[k].m)*100), pb=Math.round((vb/SM[k].m)*100);
    bars+='<div class="brow"><div class="blbl">'+SM[k].l+'</div><div style="flex:1">'
      +'<div style="display:flex;align-items:center;gap:.5rem;margin-bottom:3px"><span style="font-size:.68rem;font-family:var(--fm);color:var(--ink4);width:12px">A</span><div class="btrack" style="height:13px"><div class="bfill" style="width:'+pa+'%;background:'+scoreColor(pa)+'"><span>'+va+'</span></div></div></div>'
      +'<div style="display:flex;align-items:center;gap:.5rem"><span style="font-size:.68rem;font-family:var(--fm);color:var(--ink4);width:12px">B</span><div class="btrack" style="height:13px"><div class="bfill" style="width:'+pb+'%;background:'+scoreColor(pb)+'"><span>'+vb+'</span></div></div></div>'
      +'</div></div>';
  });
  $('cmp-result').innerHTML='<div class="cgrid">'
    +'<div class="cpanel"><h4>Report A'+(winner==='A'?' <span class="wbadge">Winner ✓</span>':'')+'</h4><div class="csc" style="color:'+scoreColor(A.score)+'">'+A.score+'<span style="font-size:.9rem;color:var(--ink4)">/100</span></div><div style="font-size:.8rem;color:var(--ink3);margin:.28rem 0 .45rem;font-family:var(--fm)">'+gradeMeta(A.score).label+'</div><div style="font-size:.84rem;font-weight:600;color:var(--ink);margin-bottom:.25rem">'+esc(A.title)+'</div><div style="font-size:.73rem;color:var(--ink4);font-family:var(--fm)">'+ago(A.timestamp)+'</div></div>'
    +'<div class="cpanel"><h4>Report B'+(winner==='B'?' <span class="wbadge">Winner ✓</span>':'')+'</h4><div class="csc" style="color:'+scoreColor(B.score)+'">'+B.score+'<span style="font-size:.9rem;color:var(--ink4)">/100</span></div><div style="font-size:.8rem;color:var(--ink3);margin:.28rem 0 .45rem;font-family:var(--fm)">'+gradeMeta(B.score).label+'</div><div style="font-size:.84rem;font-weight:600;color:var(--ink);margin-bottom:.25rem">'+esc(B.title)+'</div><div style="font-size:.73rem;color:var(--ink4);font-family:var(--fm)">'+ago(B.timestamp)+'</div></div>'
    +'</div>'
    +'<div class="dc"><h3>Section Comparison</h3>'+bars+'</div>';
}

function renderDashboard(){
  var el=$('dash-area');
  if(!HIST.length){ el.innerHTML='<div class="nodata" style="margin-top:1rem">No data yet — run some analyses first!</div>'; return; }
  var total=HIST.length;
  var avg=Math.round(HIST.reduce(function(a,b){return a+b.score;},0)/total);
  var passing=HIST.filter(function(h){return h.score>=75;}).length;
  var critical=HIST.filter(function(h){return h.score<55;}).length;
  var grades={Excellent:0,Good:0,'Needs Work':0,Incomplete:0};
  HIST.forEach(function(h){grades[h.grade]=(grades[h.grade]||0)+1;});
  var gradeColors={Excellent:'#1d6b4e',Good:'#1e4b8a','Needs Work':'#b5830a',Incomplete:'#c4410c'};
  var assignees={};
  HIST.forEach(function(h){ var a=(h.parsed&&h.parsed.recommended_assignee)||'Unknown'; assignees[a]=(assignees[a]||0)+1; });

  var gradeBars=Object.keys(grades).map(function(g){
    var cnt=grades[g], pct=Math.round((cnt/total)*100);
    return '<div class="brow"><div class="blbl">'+g+'</div><div class="btrack"><div class="bfill" style="width:'+pct+'%;background:'+gradeColors[g]+'"><span>'+pct+'%</span></div></div><div class="bnum">'+cnt+'</div></div>';
  }).join('');

  var aKeys=Object.keys(assignees).sort(function(a,b){return assignees[b]-assignees[a];}).slice(0,6);
  var assignBars=aKeys.map(function(a){
    var cnt=assignees[a], pct=Math.round((cnt/total)*100);
    return '<div class="brow"><div class="blbl">'+esc(a)+'</div><div class="btrack"><div class="bfill" style="width:'+pct+'%;background:#1e4b8a"><span>'+pct+'%</span></div></div><div class="bnum">'+cnt+'</div></div>';
  }).join('');

  var sparkItems=HIST.slice(0,20).reverse();
  var spark=sparkItems.map(function(h){
    var pct=h.score; var c=scoreColor(h.score);
    return '<div class="spk" title="'+h.score+'/100 — '+esc(h.title.slice(0,30))+'" style="height:'+pct+'%;background:'+c+'"></div>';
  }).join('');

  var recentHtml='';
  HIST.slice(0,5).forEach(function(h){
    var col=scoreColor(h.score), gm=gradeMeta(h.score);
    recentHtml+='<div class="hi" style="cursor:pointer" id="rh-'+h.id+'">'
      +'<div class="h-badge" style="border-color:'+col+';color:'+col+';width:44px;height:44px"><div class="n" style="font-size:.84rem">'+h.score+'</div></div>'
      +'<div class="h-info"><div class="h-title">'+esc(h.title)+'</div><div class="h-meta">'+ago(h.timestamp)+'</div></div>'
      +'<div class="h-grade" style="background:'+gm.bg+';color:'+gm.color+'">'+gm.label+'</div>'
      +'</div>';
  });

  el.innerHTML='<div class="sgrid">'
    +'<div class="sc"><div class="sl">Total Analyses</div><div class="sv">'+total+'</div><div class="ss">reports checked</div></div>'
    +'<div class="sc"><div class="sl">Average Score</div><div class="sv" style="color:'+scoreColor(avg)+'">'+avg+'</div><div class="ss">out of 100</div></div>'
    +'<div class="sc"><div class="sl">Passing ≥ 75</div><div class="sv" style="color:#1d6b4e">'+passing+'</div><div class="ss">'+Math.round((passing/total)*100)+'% of total</div></div>'
    +'<div class="sc"><div class="sl">Critical &lt; 55</div><div class="sv" style="color:#c4410c">'+critical+'</div><div class="ss">need rework</div></div>'
    +'</div>'
    +'<div class="dc"><h3>Score Trend (last '+Math.min(20,total)+' reports)</h3><div class="spark">'+spark+'</div><div style="display:flex;justify-content:space-between;font-size:.63rem;font-family:var(--fm);color:var(--ink4);margin-top:.25rem"><span>oldest</span><span>newest →</span></div></div>'
    +'<div style="display:grid;grid-template-columns:1fr 1fr;gap:1rem;margin-bottom:.9rem">'
      +'<div class="dc" style="margin-bottom:0"><h3>Grade Distribution</h3>'+gradeBars+'</div>'
      +'<div class="dc" style="margin-bottom:0"><h3>Recommended Assignee</h3>'+(assignBars||'<div class="nodata">No data</div>')+'</div>'
    +'</div>'
    +'<div class="dc"><h3>Recent Analyses</h3><div style="display:flex;flex-direction:column;gap:.55rem">'+recentHtml+'</div></div>';

  HIST.slice(0,5).forEach(function(h){
    var el2=$('rh-'+h.id); if(el2) el2.addEventListener('click',function(){viewHistItem(h.id);});
  });
}
